# Changelog

The complete release history, generated automatically from
[Changesets](https://github.com/changesets/changesets) every time we
publish to npm. Want only the highlights? Subscribe to
[GitHub Releases](https://github.com/kensaurus/mushi-mushi/releases) for
a curated email feed.

<div className="mt-8 space-y-12">
  {changelog.map((release) => (
    <section
      key={release.majorMinor}
      id={`v${release.majorMinor}`}
      className="border-l-2 border-neutral-200 pl-6 dark:border-neutral-800"
    >
      <header className="mb-4 flex flex-wrap items-baseline gap-3">
        <h2 className="!m-0 text-2xl font-semibold tracking-tight">
          v{release.majorMinor}
        </h2>
        {release.pending && (
          <span className="rounded-full bg-amber-100 px-2 py-0.5 text-xs font-medium text-amber-900 dark:bg-amber-900/30 dark:text-amber-300">
            upcoming
          </span>
        )}
        {release.versions?.length ? (
          <span className="text-sm text-neutral-500 dark:text-neutral-400">
            includes {release.versions.join(', ')}
          </span>
        ) : null}
      </header>

      {release.headline && (
        <p className="mb-3 text-base font-medium text-neutral-700 dark:text-neutral-200">
          {release.headline}
        </p>
      )}

      {release.highlights?.length > 0 && (
        <ul className="!my-0 list-disc space-y-1 pl-5 text-sm text-neutral-700 dark:text-neutral-300">
          {release.highlights.map((h, i) => (
            <li key={i}>
              <span className="font-medium text-neutral-900 dark:text-neutral-100">
                {h.title}
              </span>
              {h.description ? <span> — {h.description}</span> : null}
            </li>
          ))}
        </ul>
      )}
    </section>
  ))}
</div>
